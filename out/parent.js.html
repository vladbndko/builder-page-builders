/**
 * Wrap the fields into the parent
 * @function
 * @param {array} children
 */
const wrapInParent = (children) => [
  {
    type: 'tag',
    name: 'div',
    voidElement: false,
    attrs: {
      class: 'mb-3',
    },
    children,
  },
];

export default wrapInParent;
